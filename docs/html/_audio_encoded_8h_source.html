411 bool begin() {
412 LOGD(LOG_METHOD);
413 if (ptr_in==nullptr) return false;
414 dec_stream.begin(&buffer_blocking, decoder_ptr);
415 if (p_notify!=nullptr){
416 dec_stream.setNotifyAudioChange(*p_notify);

418 copier.begin(dec_stream, *ptr_in);

434 size_t readBytes(uint8_t *data, size_t len){
435 if (!active) return 0;
436 size_t result = buffer.readBytes(data, len);
437 if (auto_load && result==0){
438 refill();
439 result = buffer.readBytes(data, len);
440 }

442 if (result==0){
443 LOGW("readBytes: %d", result);

446 return result;

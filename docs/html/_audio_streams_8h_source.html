193 size_t result = 0;

319 struct DataNode {
320 int len=0;
321 uint8_t* data=nullptr;
322 DataNode() = default;
324 DataNode(void*data, int len){
325 this->len = len;
326 this->data = new uint8_t[len];

328 }

332 }
333 };

335 DynamicMemoryStream() = default;

337 this->default_buffer_size = defaultBufferSize;
338 is_loop = isLoop;
339 }
340 // Assign values from ref, clearing the original ref
341 void assign(DynamicMemoryStream &ref){
342 audio_list.swap(ref.audio_list);
343 it = ref.it;
344 total_available=ref.total_available;
345 default_buffer_size = ref.default_buffer_size;
346 alloc_failed = ref.alloc_failed;;
347 is_loop = ref.is_loop;
348 ref.clear();
349 }
350
352 virtual bool begin(){
353 clear();
354 temp_audio.resize(default_buffer_size);
355 return true;
356 }
357
358 virtual void end() {
359 clear();
360 }
361
362 virtual void setLoop(bool loop){
363 is_loop = loop;
364 }
365
366 void clear() {
367 audio_list.clear();
368 temp_audio.reset();
369 total_available = 0;
370 alloc_failed = false;
371 rewind();
372 }
373
374 size_t size(){
375 return total_available;
376 }
377
379 void rewind() {
380 it = audio_list.begin();
381 }
382
383 virtual size_t write(const uint8_t *buffer, size_t size) override {
384 DataNode node((void*)buffer, size);
385 if (node.data!=nullptr){
386 alloc_failed = false;
387 total_available += size;
388 audio_list.push_back(node);
389
390 // setup interator to point to first record
391 if (it == audio_list.end()){
392 it = audio_list.begin();
393 }
394
395 return size;
396 }
397 alloc_failed = true;
398 return 0;
399 }
400
401 virtual int availableForWrite() override {
402 return alloc_failed ? 0 : default_buffer_size;
403 }
404
405 virtual int available() override {
406 return it == audio_list.end() ? 0 : (*it).len;
407 }
408
409 virtual size_t readBytes(uint8_t *buffer, size_t length) override {
410 // provide unprocessed data
411 if (temp_audio.available()>0){
412 return temp_audio.readArray(buffer, length);
413 }
414
415 // We have no more data
416 if (it==audio_list.end()){
417 if (is_loop){
418 rewind();
419 } else {
420 // stop the processing
421 return 0;
422 }
423 }
424
425 // provide data from next node
426 DataNode node = *it;
427 int result_len = min(length, (size_t) node.len);
428 memcpy(buffer, node.data, result_len);
429 // save unprocessed data to temp buffer
430 if (node.len>length){
431 uint8_t *start = node.data+result_len;
432 int uprocessed_len = node.len - length;
433 temp_audio.writeArray(start, uprocessed_len);
434 }
435 //move to next pos
436 ++it;
437 return result_len;
438 }
439
440protected:
441 List<DataNode> audio_list;
442 List<DataNode>::Iterator it = audio_list.end();
443 size_t total_available=0;
444 int default_buffer_size=DEFAULT_BUFFER_SIZE;
445 bool alloc_failed = false;
446 RingBuffer<uint8_t> temp_audio{DEFAULT_BUFFER_SIZE};
447 bool is_loop = false;
